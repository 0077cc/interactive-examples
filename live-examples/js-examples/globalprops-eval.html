function equivalence(x,y){
  if (eval(x) === eval(y)) return "Strong";
  if (eval(x.toString()) === eval(y.toString())) return "Weak";
  return "None";
}

console.log(equivalence('2 + 2', '4'));
//expected output: Strong

console.log(equivalence('2 + 2', new String('2 + 2')));
//expected output: Weak
