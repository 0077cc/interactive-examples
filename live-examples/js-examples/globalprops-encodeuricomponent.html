var uriComp = "?x=шеллы";
var encodedComp = encodeURIComponent(uriComp);
console.log(encodedComp);
// expected output: %3Fx%3D%D1%88%D0%B5%D0%BB%D0%BB%D1%8B
var encodedUri = encodeURI(uriComp);
console.log(encodedUri);
// expected output: ?x=%D1%88%D0%B5%D0%BB%D0%BB%D1%8B

console.log(decodeURIComponent(encodedComp));
// expected output: ?x=шеллы
console.log(decodeURI(encodedComp));
// expected output: %3Fx%3Dшеллы
