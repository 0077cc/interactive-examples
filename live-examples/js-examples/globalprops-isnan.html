console.log(isNaN(undefined));
// expected output: true
console.log(isNaN(null));
// expected output: false
console.log(isNaN(37));
// expected output: false
console.log(isNaN('37'));
// expected output: false
console.log(isNaN('37ABC'));
// expected output: true
console.log(isNaN('')); // or space equivalent to zero
// expected output: true
